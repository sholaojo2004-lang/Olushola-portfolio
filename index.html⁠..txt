    <script>
        // Initialize Icons
        lucide.createIcons();

        const WHATSAPP_NUMBER = "2347044969229"; // Your formatted number

        // Sample initial items if local storage is empty
        const defaultItems = [
            { name: "Premium Virtual Number", desc: "Bulk verified numbers for communication setups.", price: "₦3,500" },
            { name: "Custom Brand Flyer", desc: "Professional CorelDRAW layout tailored to your brand.", price: "₦5,000" }
        ];

        // Load items from localStorage
        let items = JSON.parse(localStorage.getItem('storeItems')) || defaultItems;

        function renderProducts() {
            const grid = document.getElementById('productStore');
            grid.innerHTML = '';

            if (items.length === 0) {
                grid.innerHTML = `<p style="color: var(--text-muted); font-size: 0.9rem; text-align: center;">No items added yet.</p>`;
                return;
            }

            items.forEach((item, index) => {
                const productCard = document.createElement('div');
                productCard.className = 'product-card';
                
                // Encode package information cleanly for the WhatsApp API redirect
                const whatsappMessage = encodeURIComponent(`Hello Olushola! I visited your website and I want to order:\n\n📦 Item: ${item.name}\n📋 Details: ${item.desc}\n💰 Price: ${item.price}`);
                const waLink = `https://wa.me/${WHATSAPP_NUMBER}?text=${whatsappMessage}`;

                productCard.innerHTML = `
                    <div class="product-info">
                        <h3>${item.name}</h3>
                        <p>${item.desc}</p>
                        <div class="product-price">${item.price}</div>
                    </div>
                    <button class="order-btn" onclick="window.open('${waLink}', '_blank')">
                        Order
                        <i data-lucide="external-link" style="width: 14px; height: 14px;"></i>
                    </button>
                `;
                grid.appendChild(productCard);
            });
            lucide.createIcons();
        }

        // Handle Add Item Submit
        document.getElementById('adminForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('itemName').value;
            const desc = document.getElementById('itemDesc').value;
            const price = document.getElementById('itemPrice').value;

            items.push({ name, desc, price });
            localStorage.setItem('storeItems', JSON.stringify(items));
            
            renderProducts();
            this.reset();
        });

        // First Render
        renderProducts();
    </script>
